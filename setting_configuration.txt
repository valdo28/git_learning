setting email user pakai email yang didaftarkan di git
git config --global user.email 'masukkan emailmu'

konfigurasi user
git config --global user.name 'user nama anda'

adding file
git add <sebutkan namafile>
git add .

remove file
git rm --cached <sebutkan namafile>

commit (membuat message)
git commit -m "Bro tolong check, udh nambah feature"

git branch -M main

git remote add origin https://github.com/Valdo28/git_learning.git

git push -u origin main

check config
git config --global --list
